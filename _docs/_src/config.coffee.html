56    set: ( key, value )=>
57      if not key?
58        return
59      
60      if key is "set"
61        return
62      
63      if _isObject( key )
64        for _k, _v of key when key.hasOwnProperty( _k )
65          @set( _k, _v )
66        return
67      if _isObject( @[ key ] ) and _isObject( value ) and not _isArray( value ) and not _isFunction( value )
68        @[ key ] = _assignIn( true, {}, @[ key ], value )
69      else
70        @[ key ] = value
